（3）使用对象池/单例模式与在main中统一管理也没有区别吧，都是只生成了一次service和数据库：
func GetUserService() *UserService {
    once.Do(func() {
        // 确保只创建一次，修改也只需要改这里
        db := MySQLDB{}
        instance = &UserService{store: db}
    })
    return instance
}
（4） 依赖倒置方式不会创建多个数据库实例吗：
type Factory struct {
    storeProvider func() DataStore  // 传入创建函数
}

func NewFactory(provider func() DataStore) *Factory {
    return &Factory{storeProvider: provider}
}
